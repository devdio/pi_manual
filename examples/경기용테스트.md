## 기본적인 라인트레이싱
라인을 벗어나면 그자리에서 멈춘다.

```python
#-*-coding:utf-8-*-
from pyKamipi.pibot import *

kb = KamibotPi('COM8', 57600)
kb.init()

kb.turn_led(255, 0, 0)    # 빨강
MAX_SPEED = 100
MIN_SPEED = 30

try:
    while True:
        # 라인 센서 읽기
        left, center, right = kb.get_line_sensor(True)
        
        if center == 1:  # 중앙에 라인
            kb.go_forward_speed(MAX_SPEED, MAX_SPEED)
        elif left == 1:  # 왼쪽에 라인
            kb.go_forward_speed(MIN_SPEED, MAX_SPEED)  # 좌회전
        elif right == 1:  # 오른쪽에 라인
            kb.go_forward_speed(MAX_SPEED, MIN_SPEED)  # 우회전
        else:  # 라인 없음
            print("라인 없음, 정지")
            print(left, center, right)

            kb.stop()
            kb.turn_led_idx(1)  # 빨강
            
        kb.delay(0.05)
        
except KeyboardInterrupt:
    print("프로그램 중단")
finally:
    kb.close()
```
